-- SQL script to recreate the sales table in Supabase (PostgreSQL)
CREATE TABLE IF NOT EXISTS sales_table (
    id SERIAL PRIMARY KEY,
    order_id INT8,
    orders_date TIMESTAMP,
    customer_name TEXT,
    customer_email TEXT,
    amount TEXT, -- Keep as text if it contains symbols, or NUMERIC for math
    region TEXT,
    status TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
